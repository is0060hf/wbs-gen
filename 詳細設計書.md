- **React**: 18+ (Server/Client Components)
- **状態管理**: Context API + useReducer（Client Components内）
- **UI**: Tailwind CSS (コアユーティリティのみ)
- **アイコン**: lucide-react
- **日付処理**: ブラウザネイティブDate API
- **チャート**: Recharts（ガントチャート用）

  is_critical?: boolean;         // クリティカルパスフラグ
  notes?: string;                // メモ

## データ正規化とデフォルト値処理

```typescript
// app/lib/wbs-utils.ts

export function normalizeWBSTask(task: Partial<WBSTask>): WBSTask {
  // 終了日の自動計算
  const calculateEndDate = (start: string, duration: number): string => {
    const startDate = new Date(start);
    const endDate = new Date(startDate);
    endDate.setDate(startDate.getDate() + Math.ceil(duration) - 1);
    return endDate.toISOString().split('T')[0];
  };

  // デフォルト値の設定
  const normalized: WBSTask = {
    id: task.id || generateTaskId(),
    wbs_code: task.wbs_code || '1',
    name: task.name || '新規タスク',
    priority: task.priority || 'Must',
    start: task.start || new Date().toISOString().split('T')[0],
    duration_days: task.duration_days || 1,
    
    // オプショナルフィールドのデフォルト値
    progress: task.progress ?? 0,
    status: task.status ?? 'Not Started',
    dependencies: task.dependencies || [],
    
    // 終了日の処理（値があれば使用、なければ計算）
    end: task.end || calculateEndDate(task.start!, task.duration_days || 1),
    
    // その他のオプショナルフィールド
    description: task.description,
    assignee: task.assignee,
    buffer: task.buffer,
    is_critical: task.is_critical ?? false,
    notes: task.notes,
    
    // 子タスクの再帰処理
    children: task.children?.map(child => normalizeWBSTask(child))
  };

  return normalized;
}

// インポート時の正規化処理
export function normalizeImportedProject(data: any): Project {
  // project_infoの正規化
  const projectInfo: ProjectInfo = {
    name: data.project_info?.name || 'インポートされたプロジェクト',
    version: data.project_info?.version || '1.0.0',
    pmbok_version: data.project_info?.pmbok_version || '7',
    description: data.project_info?.description || '',
    created_at: data.project_info?.created_at || new Date().toISOString().split('T')[0],
    updated_at: data.project_info?.updated_at || new Date().toISOString().split('T')[0]
  };

  // WBSタスクの正規化
  const wbs = (data.wbs || []).map((task: any) => normalizeWBSTask(task));

  return {
    project_info: projectInfo,
    wbs: wbs
  };
}

// タスクID生成
function generateTaskId(): string {
  return `T-${Date.now()}-${Math.random().toString(36).substr(2, 9)}`;
}
```

// app/components/features/ImportExport.tsx
'use client'

import { useWBS } from '@/hooks/useWBS'

export function ImportExportButtons() {
  const { dispatch } = useWBS()

  const handleImport = async (event: React.ChangeEvent<HTMLInputElement>) => {
    const file = event.target.files?.[0]
    if (!file) return

    try {
      const text = await file.text()
      const data = JSON.parse(text)
      
      // データの正規化
      const normalizedProject = normalizeImportedProject(data)
      
      dispatch({
        type: 'IMPORT_PROJECT',
        payload: normalizedProject
      })
      
      toast.success('プロジェクトをインポートしました')
    } catch (error) {
      toast.error('インポートに失敗しました。JSONファイルを確認してください。')
    }
  }

  const handleExport = () => {
    const project = getCurrentProject() // Context から現在のプロジェクトを取得
    
    // エクスポート時は計算済みの全フィールドを含める
    const exportData = {
      ...project,
      project_info: {
        ...project.project_info,
        updated_at: new Date().toISOString().split('T')[0]
      }
    }
    
    const blob = new Blob([JSON.stringify(exportData, null, 2)], {
      type: 'application/json'
    })
    
    const url = URL.createObjectURL(blob)
    const a = document.createElement('a')
    a.href = url
    a.download = `wbs-${project.project_info.name}-${Date.now()}.json`
    a.click()
    URL.revokeObjectURL(url)
  }

  return (
    <>
      <input
        type="file"
        accept=".json"
        onChange={handleImport}
        className="hidden"
        id="import-file"
      />
      <label
        htmlFor="import-file"
        className="cursor-pointer px-4 py-2 bg-blue-500 text-white rounded hover:bg-blue-600"
      >
        JSONインポート
      </label>
      <button
        onClick={handleExport}
        className="px-4 py-2 bg-green-500 text-white rounded hover:bg-green-600"
      >
        JSONエクスポート
      </button>
    </>
  )
}

// app/lib/schedule-calculator.ts

export function recalculateSchedule(tasks: WBSTask[]): WBSTask[] {
  // タスクマップの作成
  const taskMap = new Map<string, WBSTask>()
  const flattenTasks = (taskList: WBSTask[]) => {
    taskList.forEach(task => {
      taskMap.set(task.id, task)
      if (task.children) flattenTasks(task.children)
    })
  }
  flattenTasks(tasks)

  // 依存関係に基づく開始日の再計算
  const calculateDependentStart = (task: WBSTask): string => {
    if (!task.dependencies || task.dependencies.length === 0) {
      return task.start
    }

    let latestEnd = new Date(task.start)
    
    task.dependencies.forEach(depId => {
      const depTask = taskMap.get(depId)
      if (depTask && depTask.end) {
        const depEnd = new Date(depTask.end)
        // 依存タスクの翌日を開始日とする
        depEnd.setDate(depEnd.getDate() + 1)
        if (depEnd > latestEnd) {
          latestEnd = depEnd
        }
      }
    })

    return latestEnd.toISOString().split('T')[0]
  }

  // 全タスクの再計算
  const recalculateTask = (task: WBSTask): WBSTask => {
    const newStart = calculateDependentStart(task)
    const newEnd = calculateEndDate(newStart, task.duration_days)
    
    return {
      ...task,
      start: newStart,
      end: newEnd,
      children: task.children?.map(child => recalculateTask(child))
    }
  }

  return tasks.map(task => recalculateTask(task))
}

// app/lib/validation.ts

export interface ValidationResult {
  isValid: boolean;
  errors: ValidationError[];
}

export interface ValidationError {
  taskId: string;
  field: string;
  message: string;
}

export function validateWBSTask(task: WBSTask, allTasks: WBSTask[]): ValidationResult {
  const errors: ValidationError[] = [];

  // 必須フィールドのチェック
  if (!task.name || task.name.trim() === '') {
    errors.push({
      taskId: task.id,
      field: 'name',
      message: 'タスク名は必須です'
    });
  }

  // 日付の妥当性チェック
  const startDate = new Date(task.start);
  if (isNaN(startDate.getTime())) {
    errors.push({
      taskId: task.id,
      field: 'start',
      message: '開始日が不正です'
    });
  }

  // 期間の妥当性チェック
  if (task.duration_days <= 0) {
    errors.push({
      taskId: task.id,
      field: 'duration_days',
      message: '期間は1日以上である必要があります'
    });
  }

  // 循環依存のチェック
  if (task.dependencies && hasCircularDependency(task, allTasks)) {
    errors.push({
      taskId: task.id,
      field: 'dependencies',
      message: '循環依存が検出されました'
    });
  }

  return {
    isValid: errors.length === 0,
    errors
  };
}

## 実装指示（更新版）

このプロンプトに基づいて、以下の点に特に注意して実装してください：

1. **柔軟なインポート機能**
   - `progress`と`status`は省略可能（デフォルト値で補完）
   - `dependencies`フィールドで統一（`predecessors`も受け入れ可能）
   - 終了日は省略時に自動計算

2. **エクスポート機能**
   - すべての計算済みフィールドを含めて出力
   - 人間が読みやすい整形済みJSON

3. **リアルタイム計算**
   - タスクの開始日や期間を変更したら、依存タスクの日程を自動調整
   - 終了日は常に自動計算で最新化

4. **エラーハンドリング**
   - 不完全なデータでもできる限り読み込む
   - エラーは分かりやすくユーザーに表示

これにより、既存のWBSデータとの互換性を保ちながら、高機能なWBS管理システムを実現できます。
